class Cell:
    vector <Cell*> voisins;
    Pion* occupied;
  public:
    void Cell: occupied{nullptr} ()

    void set_voisins(vector <Cell*> new_voisins) {
        this->voisins = new_voisins;
    }

    void accept_pion(Pion* new_pion, Cell* old_cell) {
        this->occupied = new_pion;
        old_cell->occupied = nullptr;
    }
