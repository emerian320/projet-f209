class Cell:
    	vector <Cell*> neighbours
    	Pawn* occupied
    public:
	Cell(): occupied{nullptr} {}
    	void setNeighbours(vector <Cell*> newNeighbours) {
    	    this->neighbours = newNeighbours
    	}

    	void acceptPawn(Pawn* newPawn, Cell* oldCell) {
    	    this->occupied = newPawn
    	    oldCell->occupied = nullptr
    	}
